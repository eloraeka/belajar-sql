Belajar syntax SQL

Urutan lengkap dari CREATE DATABASE sampai TRANSACTION · semua contoh bisa dijalankan.

mysql — belajar
mysql>
01CREATE DATABASE

Membuat database baru. Nama database harus unik dan mengikuti aturan penamaan.

DDL
CREATE DATABASE sekolah_ku;
02USE DATABASE

Memilih database yang akan digunakan untuk operasi berikutnya.

DDL
USE sekolah_ku;
03CREATE TABLE

Membuat tabel baru beserta kolom-kolomnya.

DDL
CREATE TABLE siswa (
  id INT,
  nama VARCHAR(50),
  umur INT
);
04Data Type

Tipe data menentukan jenis nilai yang bisa disimpan di kolom.

DDL
CREATE TABLE contoh_tipe (
  id INT,
  nama VARCHAR(100),
  tanggal_lahir DATE,
  gaji DECIMAL(10,2),
  aktif BOOLEAN
);
05Constraint

Aturan validasi data: PRIMARY KEY, FOREIGN KEY, NOT NULL, UNIQUE, DEFAULT, CHECK.

DDL
CREATE TABLE mahasiswa (
  npm CHAR(10) PRIMARY KEY,
  nama VARCHAR(50) NOT NULL,
  jurusan VARCHAR(30) DEFAULT 'Informatika',
  usia INT CHECK (usia >= 17),
  email VARCHAR(100) UNIQUE
);
06REFERENCES

Menghubungkan foreign key ke primary key di tabel lain.

DDL
CREATE TABLE nilai (
  id INT PRIMARY KEY,
  npm CHAR(10),
  mata_kuliah VARCHAR(50),
  FOREIGN KEY (npm) REFERENCES mahasiswa(npm)
);
07ON DELETE

Aksi saat baris di tabel induk dihapus: CASCADE, SET NULL, RESTRICT.

DDL
CREATE TABLE nilai_ondelete (
  id INT PRIMARY KEY,
  npm CHAR(10),
  FOREIGN KEY (npm) REFERENCES mahasiswa(npm)
    ON DELETE CASCADE
);
08ON UPDATE CASCADE

Saat primary key di tabel induk diubah, foreign key di tabel anak ikut berubah.

DDL
CREATE TABLE nilai_onupdate (
  id INT PRIMARY KEY,
  npm CHAR(10),
  FOREIGN KEY (npm) REFERENCES mahasiswa(npm)
    ON UPDATE CASCADE
);
09ALTER TABLE

Mengubah struktur tabel: tambah kolom, ubah tipe, tambah constraint.

DDL
ALTER TABLE mahasiswa
ADD COLUMN alamat TEXT;
10DROP TABLE

Menghapus tabel beserta seluruh data dan strukturnya.

DDL
DROP TABLE IF EXISTS nilai_ondelete;
11TRUNCATE TABLE

Menghapus semua data di tabel tapi struktur tetap ada.

DDL
TRUNCATE TABLE mahasiswa;
12INSERT

Menambah baris data ke dalam tabel.

DML
INSERT INTO mahasiswa (npm, nama, jurusan, usia, email)
VALUES ('M001', 'Budi', 'Informatika', 20, 'budi@mail.com');
13SELECT

Mengambil data dari tabel.

DML
SELECT * FROM mahasiswa;
14WHERE

Menyaring baris berdasarkan kondisi.

DML
SELECT * FROM mahasiswa
WHERE usia >= 20;
15LIKE & REGEXP

Pencocokan pola teks. LIKE pakai % dan _, REGEXP pakai regex.

DML
SELECT * FROM mahasiswa
WHERE nama LIKE 'B%';
16String Function

Fungsi untuk mengolah teks: CONCAT, LEFT, UPPER, LENGTH, dll.

DML
SELECT UPPER(nama) AS nama_besar,
       LENGTH(nama) AS panjang
FROM mahasiswa;
17Date Function

Fungsi untuk tanggal: CURDATE, DATEDIFF, YEAR, MONTH, dll.

DML
SELECT CURDATE() AS hari_ini,
       YEAR(CURDATE()) AS tahun;
18ORDER BY

Mengurutkan hasil query.

DML
SELECT * FROM mahasiswa
ORDER BY usia DESC;
19Aggregate Function

COUNT, SUM, AVG, MAX, MIN.

DML
SELECT COUNT(*) AS total,
       AVG(usia) AS rata_usia
FROM mahasiswa;
20GROUP BY

Mengelompokkan data untuk agregasi per kelompok.

DML
SELECT jurusan, COUNT(*) AS jumlah
FROM mahasiswa
GROUP BY jurusan;
21HAVING

Menyaring hasil GROUP BY.

DML
SELECT jurusan, COUNT(*) AS jumlah
FROM mahasiswa
GROUP BY jurusan
HAVING COUNT(*) > 0;
22JOIN

Menggabungkan tabel berdasarkan relasi.

DML
SELECT mahasiswa.nama, nilai.mata_kuliah
FROM mahasiswa
INNER JOIN nilai ON mahasiswa.npm = nilai.npm;
23UPDATE

Mengubah nilai pada baris yang sudah ada.

DML
UPDATE mahasiswa
SET usia = 21
WHERE npm = 'M001';
24DELETE

Menghapus baris berdasarkan kondisi.

DML
DELETE FROM mahasiswa
WHERE npm = 'M001';
25TRANSACTION

Membungkus beberapa query dalam satu unit kerja.

DML
START TRANSACTION;
UPDATE mahasiswa SET usia = 22 WHERE npm = 'M002';
ROLLBACK;

Uji pemahaman tentang seluruh materi.
